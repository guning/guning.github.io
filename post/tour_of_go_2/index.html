Go语言的协程和通道（channels）

 Posted on November 16, 2018  |  4 minutes (776 words)

go的协程概念声名远扬，究竟何许物也，这章就来看看。

Introduction

//调用方式
go funcName

协程的启动方式如上，十分简单。

channels

与其他语言的线程概念类似，协程共享内存，故会有数据同步上的问题，除了通过sync关键字之外，还可以使用channels类型来做buffer（类似队列结构）来进行数据通信，保证数据一致性。

package main

import "fmt"

func sum(s []int, c chan int) {
	sum := 0
	for _, v := range s {
		sum += v
	}
	c <- sum // send sum to c
}

func main() {
	s := []int{7, 2, 8, -9, 4, 0}

	c := make(chan int)
	go sum(s[:len(s)/2], c)
	go sum(s[len(s)/2:], c)
	x, y := <-c, <-c // receive from c

	fmt.Println(x, y, x+y)
}
  • channels可以设置长度，传参在make的第二个参数。这个时候如果写入长度超过定义的程度，会报panic
  • channels也可以在发送端关闭，如果向关闭的channels发数据，会造成panic
  • select语法：如下，会自动选择准备好的channels，其default条件适合用来做延迟，避免过度占用系统资源

execise之二叉树节点值比较

实现方式：遍历两棵待比较的二叉树，并对数据进行排序吗，再逐项进行比较。涉及到的算法有：二叉树遍历，二分法查找

package main

import (
	"fmt"
	"time"
)

func main() {
	tick := time.Tick(100 * time.Millisecond)
	boom := time.After(500 * time.Millisecond)
	for {
		select {
		case <-tick:
			fmt.Println("tick.")
		case <-boom:
			fmt.Println("BOOM!")
			return
		default:
			fmt.Println("    .")
			time.Sleep(50 * time.Millisecond)
		}
	}
}
package main

import (
	"fmt"
	"golang.org/x/tour/tree"
	"math"
)

type TmpData []int

func (td *TmpData) insertSort(d int) {
	//插入排序
	tmpData := []int(*td)
	if len(tmpData) == 0 {
		*td = append(tmpData, d)
		return
	}
	tmpData = append(tmpData, d)
	for i := len(tmpData) - 1; i > 0; i-- {
		if tmpData[i] < tmpData[i-1] {
			tmpData[i], tmpData[i-1] = tmpData[i-1], tmpData[i]
		} else {
			break
		}
	}
	*td = TmpData(tmpData)
}

func (td TmpData) search(d int) bool {
	//二分法查找
	tmpData := []int(td)
	startIndex := 0
	endIndex := len(tmpData) - 1
	if tmpData[endIndex] == d {
		return true
	}
	if d > tmpData[endIndex] {
		return false
	}
	var nowIndex int
	for {
		if endIndex-startIndex <= 1 {
			if tmpData[endIndex] == d || tmpData[startIndex] == d {
				return true
			}
			return false
		}
		nowIndex = int(math.Floor(float64((endIndex + startIndex) / 2)))
		if tmpData[nowIndex] == d {
			return true
		} else if tmpData[nowIndex] > d {
			endIndex = nowIndex
		} else if tmpData[nowIndex] < d {
			startIndex = nowIndex
		}
	}
}

// Walk walks the tree t sending all values
// from the tree to the channel ch.
func Walk(t *tree.Tree, ch chan int) {
	if t == nil {
		return
	}
	ch <- t.Value
	Walk(t.Left, ch)
	Walk(t.Right, ch)
}

// Same determines whether the trees
// t1 and t2 contain the same values.
func Same(t1, t2 *tree.Tree) bool {
	ch1 := make(chan int)
	go Walk(t1, ch1)
	ch2 := make(chan int)
	go Walk(t2, ch2)
	var tmpData1 TmpData
	var tmpData2 TmpData
	for i := 0; i < 10; i++ {
		tmpData1.insertSort(<-ch1)
		tmpData2.insertSort(<-ch2)
	}
	for _, v := range tmpData1 {
		if !tmpData2.search(v) {
			return false
		}
	}
	return true
}

func main() {
	fmt.Println(Same(tree.New(1), tree.New(1)))
	fmt.Println(Same(tree.New(1), tree.New(2)))
}

补充

  • 如果我们只想在同一个时间内让一个协程获取到共享的变量，也就是给数据加互斥锁，相关包名叫sync.Mutex，包含Lock，包含Unlock，示例：
package main

import (
	"fmt"
	"sync"
	"time"
)

// SafeCounter is safe to use concurrently.
type SafeCounter struct {
	v   map[string]int
	mux sync.Mutex
}

// Inc increments the counter for the given key.
func (c *SafeCounter) Inc(key string) {
	c.mux.Lock()
	// Lock so only one goroutine at a time can access the map c.v.
	c.v[key]++
	c.mux.Unlock()
}

// Value returns the current value of the counter for the given key.
func (c *SafeCounter) Value(key string) int {
	c.mux.Lock()
	// Lock so only one goroutine at a time can access the map c.v.
	defer c.mux.Unlock()
	return c.v[key]
}

func main() {
	c := SafeCounter{v: make(map[string]int)}
	for i := 0; i < 1000; i++ {
		go c.Inc("somekey")
	}

	time.Sleep(time.Second)
	fmt.Println(c.Value("somekey"))
}

package main

import (
	"fmt"
	"sync"
)

type HasFetchMap struct {
	urls map[string]bool
	mux  sync.Mutex
}

type Fetcher interface {
	// Fetch returns the body of URL and
	// a slice of URLs found on that page.
	Fetch(url string) (body string, urls []string, err error)
}

// Crawl uses fetcher to recursively crawl
// pages starting with url, to a maximum of depth.
func Crawl(url string, depth int, fetcher Fetcher, fetchedMap HasFetchMap) {
	// TODO: Fetch URLs in parallel.
	// TODO: Don't fetch the same URL twice.
	// This implementation doesn't do either:
	if depth <= 0 {
		return
	}
	fetchedMap.mux.Lock()
	if fetchedMap.urls[url] {
	    fetchedMap.mux.Unlock()
	    return
	}
	fetchedMap.urls[url] = true
	fetchedMap.mux.Unlock()
	body, urls, err := fetcher.Fetch(url)
	if err != nil {
		fmt.Println(err)
		return
	}
	fmt.Printf("found: %s %q\n", url, body)
	for _, u := range urls {
	    Crawl(u, depth-1, fetcher, fetchedMap)
    }
	return
}

func main() {
    fetchedMap := HasFetchMap{urls: make(map[string]bool)}
	Crawl("https://golang.org/", 4, fetcher, fetchedMap)
}
