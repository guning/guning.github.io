Jewels and Stones

func numJewelsInStones(J string, S string) int {
    ss := list.New()
    for i := 0; i < len(S); i++ {
        ss.PushBack(S[i])
    }

    cnt := 0
    var next *list.Element
    for _,v := range J {
        for e := ss.Front(); e!= nil; e = next {
            next = e.Next()
            if uint8(v) == e.Value.(uint8) {
                cnt = cnt + 1
                ss.Remove(e)
            }
        }
    }
    return cnt
}
