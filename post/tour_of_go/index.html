package main

import (
	"golang.org/x/tour/wc"
	"strings"
)

// 统计每个单词出现的次数
func WordCount(s string) map[string]int {
	words := strings.Fields(s)
    res := make(map[string]int)
    for _, v := range words {
        res[v] = res[v] + 1
    }
    return res
}

func main() {
	wc.Test(WordCount)
}
package main

import "fmt"

// 闭包生成斐泊那砌数列
// fibonacci is a function that returns
// a function that returns an int.
func fibonacci() func() int {
    forward1 := -1
    forward2 := 1
    return func() int {
        tmp := forward1 + forward2
        forward1 = forward2
        forward2 = tmp
        return tmp
    }
}

func main() {
    f := fibonacci()
    for i := 0; i < 10; i++ {
        fmt.Println(f())
    }
}
