排序算法的实现及其复杂度

一、排序算法类型及其复杂度

稳定性是指：当待排序序列出现相同元素时，排序前后对应元素的相对位置有没有发生变化

快速排序 O(nlog2n) O(n2) O(nlog2n) O(nlog2n) 不稳定
插入排序 O(n2) O(n2) O(n) O(1) 稳定

二、具体的实现思路

1. 冒泡排序

func Bubbling(a []int) []int {
    for i := 0; i < len(a); i++ {
        for j := 0; j < len(a) - 1 - i; j++ {
            if a[j] > a[j+1] {
                a[j], a[j+1] = a[j+1], a[j]
            }
        }
    }
    return a
}

2. 快速排序

从第一个元素开始，找到元素在序列的位置，该位置将序列分为两部分，一部分比该元素小，另一部分比该元素大，递归进行该过程，即可解得答案。

func Quick(a []int) []int {
    depart(&a, 0, len(a) - 1)
    return a
}

func depart(a *[]int, st, ed) {
    if ed - st < 1 {
        return
    }
    //记录初始的起止位置
    rSt := st
    rEd := ed

    //从前往后，从后往前反复找到分隔该序列的元素，并转移到指定位置
    for st!=ed {
        for ed > st {
            if (*a)[st] > (*a)[ed] {
                (*a)[st], (*a)[ed] = (*a)[ed], (*a)[st]
                st++
                break
            }
            ed--
        }

        for st < ed {
            if (*a)[st] > (*a)[ed] {
                (*a)[st], (*a)[ed] = (*a)[ed], (*a)[st]
                ed--
                break
            }
            st++
        }
    }
    //此时st==ed为分隔位置下标
    depart(a, rSt, st - 1)
    depart(a, st + 1, rEd)
}

3. 插入排序

func Insert(a []int) []int {
    for i := 1; i < len(a); i++ {
        for j := i; j >= 1; j-- {
            if a[j] < a[j-1] {
                a[j], a[j-1] = a[j-1], a[j]
            }
        }
    }
    return a
}

4. 希尔排序

希尔排序是优化的插入排序，通过优化插入排序的步进值来达到目的

func Shell(a []int) []int {
    length := len(a)

    for gap := length / 2; gap >= 1; gap = gap / 2 {
        for i := gap; i < length; i++ {
            for j := i; j-gap >= 0 && a[j] < a[j-gap]; j = j - gap {
                    a[j], a[j-gap] = a[j-gap], a[j]
            }
        }
    }
    return a
}

5. 选择排序

每趟循环都在剩余序列中选择最小/最大的元素，插入到相应位置

func Select(a []int) []int {
    for i := 0; i < len(a); i++ {
        //找最小的下标
        min := i
        for j := i + 1; j < len(a); j++ {
            if a[min] > a[j] {
                    min = j
            }
        }
        if i != min {
            a[i], a[min] = a[min], a[i]
        }
    }
    return a
}

6. 堆排序

每次循环构造最小/最大堆（子结点总是小于/大于其父结点的二叉树），取根结点即为当前序列的最小/最大值（于选择排序思路一样，只是通过抽象出堆的概念，减少了交换的次数）

import "math"

func Stack(a []int) []int {
    for i := len(a); i > 0; i-- {
        getMaxStack(&a, i)
        a[0], a[i-1] = a[i-1], a[0]
    }
    return a
}

func getMaxStack(a *[]int, num int) {
    deep := int(math.Log2(float64(num)))

    for i := deep; i > 0; i-- {
        tmp := int(math.Pow(2, float64(i-1)))
        for j := 1; j <= tmp; j++ {
            checkMaxStack(a, tmp+j-2, num)
        }
    }

}

func checkMaxStack(a *[]int, f int, num int) {
    c1 := 2*f + 1
    c2 := 2*f + 2
    if c1 < num {
        if (*a)[f] < (*a)[c1] {
            (*a)[f], (*a)[c1] = (*a)[c1], (*a)[f]
        }
    }

    if c2 < num {
        if (*a)[f] < (*a)[c2] {
            (*a)[f], (*a)[c2] = (*a)[c2], (*a)[f]
        }
    }
}

7. 归并排序

对每个子序列进行拆分排序，再对结果进行归集，需要额外的空间

func Merge(a []int) []int {
        a = merge(a)
        return a
}

func merge(a []int) []int {
    if len(a) == 1 {
        return a
    }
    var res []int
    st := 0
    ed := len(a) - 1
    divide := (ed - st + 1) / 2
    left := merge(a[st:divide])
    right := merge(a[divide:])
    i := 0
    j := 0
    for i < len(left) || j < len(right) {
        if i >= len(left) {
            for t := j; t < len(right); t++ {
                res = append(res, right[t])
            }
            j = len(right)
            break
        }
        if j >= len(right) {
            for t := i; t < len(left); t++ {
                res = append(res, left[t])
            }
            i = len(left)
            break
        }
        if left[i] < right[j] {
            res = append(res, left[i])
            i++
        } else {
            res = append(res, right[j])
            j++
        }
    }
    return res
}

8. 计数排序

找到序列中最大的数max，新建一个长度为max+1的数组，下标表示序列中的数，值表示该数出现的次数，统计计数后对该数组进行循环，填值即可（缺点很明显，出现浮点数和负数的时候，需要特殊处理，否则无法使用该算法）

func Count(a []int) []int {
    max := a[0]
    for i := 1; i < len(a); i++ {
        if max < a[i] {
                max = a[i]
        }
    }
    tmp := make([]int, max+1)
    for i := 0; i < len(a); i++ {
        tmp[a[i]]++
    }

    var res []int
    for i := 0; i < len(tmp); i++ {
        for tmp[i] != 0 {
            res = append(res, i)
            tmp[i]--
        }
    }

    return res
}

9. 桶排序

按照某种规则对待排序序列进行分桶，不同桶内分别排序后进行拼接。适用于数值特征明显的序列。

const BUCKETNUM int = 5

func Bucket(a []int) []int {
    min := a[0]
    max := a[0]
    for _, v := range a {
        if min > v {
            min = v
        }
        if max < v {
            max = v
        }
    }

    buckets := make([][]int, selectBucket(max, min)+1)

    for _, v := range a {
        index := selectBucket(v, min)
        buckets[index] = append(buckets[index], v)
    }

    var res []int
    for _, v := range buckets {
        if len(v) > 0 {
            //每个桶进行插入排序
            newV := Insert(v)
            res = combine(res, newV)
        }
    }
    return res
}

func combine(a, b []int) []int {
    for _, v := range b {
        a = append(a, v)
    }
    return a
}

func selectBucket(v, min int) int {
    return (v - min) - ((v-min)%BUCKETNUM)/BUCKETNUM
}

10. 基数排序

按照数值的位数进行排序，先使低位有序，再使高位有序（限整数）

func Radix(a []int) []int {
    max := a[0]
    for _, v := range a {
        if max < v {
            max = v
        }
    }
    bitNum := getBit(max)
    for i := 1; i <= bitNum; i++ {
        dict := make([][]int, 10)
        for _, v := range a {
            index := v % (10 * i)
            dict[index] = append(dict[index], v)
        }
        cnt := 0
        for _, v := range dict {
            for _, vv := range v {
                    a[cnt] = vv
                    cnt++
            }
        }
    }
    return a
}

//获取位数
func getBit(num int) int {
    if num == 0 {
        return 0
    }
    return 1 + getBit(num/10)
}
